        class CalcAIV17Engine {

            constructor() {

                this.version = "1.7";

                this.valueTypes = {

                    UNIT_PRICE: 'unit_price',

                    QUANTITY: 'quantity',

                    TOTAL_MONEY: 'total_money',

                    WALLET_MONEY: 'wallet_money',

                    EXPENSE: 'expense',

                    DISCOUNT_PERCENT: 'discount_percent',

                    PEOPLE_COUNT: 'people_count'

                };

                

                this.calculationTypes = {

                    PURCHASE_TOTAL: 'purchase_total',

                    CHANGE_CALCULATION: 'change_calculation',

                    REMAINING_MONEY: 'remaining_money',

                    SPLIT_BILL: 'split_bill',

                    DISCOUNT_PRICE: 'discount_price',

                    TOTAL_SUM: 'total_sum'

                };

            }

            // AIモデル自動選択機能

            autoSelectModel(input) {

                const complexity = this.analyzeComplexity(input);

                

                if (complexity.isSimple) {

                    return 'speed'; // Calc-1o

                } else if (complexity.isComplex) {

                    return 'thinking'; // Calc-Thinking

                } else {

                    return 'normal'; // Calc-1

                }

            }

            analyzeComplexity(input) {

                const simplePatterns = [

                    /^\d+\s*[+\-×÷]\s*\d+/,

                    /^簡単/,

                    /^計算/

                ];

                

                const complexPatterns = [

                    /複雑|詳しく|詳細|思考|考え|なぜ|理由/,

                    /割り勘.*人/,

                    /割引.*パーセント/,

                    /複数.*計算/

                ];

                const isSimple = simplePatterns.some(pattern => pattern.test(input));

                const isComplex = complexPatterns.some(pattern => pattern.test(input));

                return { isSimple, isComplex };

            }

            // 🔥 個数単価問題完全解決：新検出エンジン

            detectValues(input) {

                console.log('=== CalcAI v1.7 個数単価問題解決開始 ===');

                console.log('入力:', input);

                

                const detectedValues = [];

                

                // Step 1: 財布金額の検出（最優先）

                const walletPattern = /財布[にで]?[¥$€£]?([\d,]+)/;

                const walletMatch = input.match(walletPattern);

                if (walletMatch) {

                    const value = parseFloat(walletMatch[1].replace(/,/g, ''));

                    detectedValues.push({

                        value: value,

                        original: `¥${value.toLocaleString()}`,

                        currency: '¥',

                        type: this.valueTypes.WALLET_MONEY,

                        context: '財布の所持金',

                        position: input.indexOf(walletMatch[0]),

                        priority: 1.0

                    });

                    console.log('財布金額検出:', value);

                }

                

                // Step 2: 「1つ¥300」パターンの検出（単価）

                const unitPricePattern = /(\d+)つ[¥$€£]([\d,]+)/;

                const unitPriceMatch = input.match(unitPricePattern);

                if (unitPriceMatch) {

                    const value = parseFloat(unitPriceMatch[2].replace(/,/g, ''));

                    detectedValues.push({

                        value: value,

                        original: `¥${value.toLocaleString()}`,

                        currency: '¥',

                        type: this.valueTypes.UNIT_PRICE,

                        context: '商品の単価',

                        position: input.indexOf(unitPriceMatch[0]),

                        priority: 0.9

                    });

                    console.log('単価検出:', value);

                }

                

                // Step 3: 「4つ買います」パターンの検出（実際の購入個数）

                const actualQuantityPattern = /(\d+)つ買/;

                const actualQuantityMatch = input.match(actualQuantityPattern);

                if (actualQuantityMatch) {

                    const value = parseInt(actualQuantityMatch[1]);

                    detectedValues.push({

                        value: value,

                        original: value.toString(),

                        currency: '',

                        type: this.valueTypes.QUANTITY,

                        context: '実際の購入個数',

                        position: input.indexOf(actualQuantityMatch[0]),

                        priority: 0.9

                    });

                    console.log('実際の購入個数検出:', value);

                }

                

                // Step 4: 通常の金額パターン（¥で始まる）

                const moneyPattern = /[¥$€£]([\d,]+)/g;

                let moneyMatch;

                while ((moneyMatch = moneyPattern.exec(input)) !== null) {

                    const value = parseFloat(moneyMatch[1].replace(/,/g, ''));

                    const position = moneyMatch.index;

                    

                    // 既に検出済みかチェック

                    const alreadyDetected = detectedValues.some(v =>

                        Math.abs(v.position - position) < 10 && v.value === value

                    );

                    

                    if (!alreadyDetected) {

                        const beforeText = input.substring(Math.max(0, position - 20), position);

                        const afterText = input.substring(position, Math.min(input.length, position + 20));

                        

                        let type = this.valueTypes.TOTAL_MONEY;

                        let context = '金額';

                        

                        // 財布の文脈

                        if (/財布|所持|手持ち|もって|ある/.test(beforeText)) {

                            type = this.valueTypes.WALLET_MONEY;

                            context = '所持金';

                        }

                        // 単価の文脈（「1つ¥300」以外の単価表現）

                        else if (/1つ|1個|単価|値段|価格/.test(beforeText) || /の商品|の品物/.test(afterText)) {

                            type = this.valueTypes.UNIT_PRICE;

                            context = '単価';

                        }

                        

                        detectedValues.push({

                            value: value,

                            original: moneyMatch[0],

                            currency: moneyMatch[0][0],

                            type: type,

                            context: context,

                            position: position,

                            priority: 0.7

                        });

                        console.log('通常金額検出:', value, 'タイプ:', type);

                    }

                }

                

                // Step 5: 数値パターン（個数候補）

                const numberPattern = /(\d+)/g;

                let numberMatch;

                while ((numberMatch = numberPattern.exec(input)) !== null) {

                    const value = parseInt(numberMatch[1]);

                    const position = numberMatch.index;

                    

                    // 既に検出済みかチェック

                    const alreadyDetected = detectedValues.some(v =>

                        Math.abs(v.position - position) < 5 && v.value === value

                    );

                    

                    if (!alreadyDetected && value < 100) {

                        const beforeText = input.substring(Math.max(0, position - 15), position);

                        const afterText = input.substring(position, Math.min(input.length, position + 15));

                        

                        // 個数の文脈判定

                        if (/つ|個|本|枚/.test(afterText) && !/¥/.test(beforeText + afterText)) {

                            detectedValues.push({

                                value: value,

                                original: value.toString(),

                                currency: '',

                                type: this.valueTypes.QUANTITY,

                                context: '個数',

                                position: position,

                                priority: 0.6

                            });

                            console.log('個数検出:', value);

                        }

                    }

                }

                

                // 重複除去と優先度ソート

                const finalValues = this.removeDuplicatesAndSort(detectedValues);

                console.log('最終検出結果:', finalValues);

                console.log('=== 検出完了 ===');

                

                return finalValues;

            }

            

            // 重複除去と優先度ソート

            removeDuplicatesAndSort(values) {

                const uniqueValues = [];

                const seen = new Set();

                

                // 優先度順にソート

                values.sort((a, b) => b.priority - a.priority);

                

                values.forEach(value => {

                    const key = `${value.type}_${value.value}`;

                    if (!seen.has(key)) {

                        seen.add(key);

                        uniqueValues.push(value);

                    }

                });

                

                return uniqueValues;

            }

            determineCalculationType(input, values) {

                console.log('=== 計算タイプ判定開始 ===');

                console.log('入力:', input);

                console.log('検出値:', values);

                

                const hasWallet = values.some(v => v.type === this.valueTypes.WALLET_MONEY);

                const hasUnitPrice = values.some(v => v.type === this.valueTypes.UNIT_PRICE);

                const hasQuantity = values.some(v => v.type === this.valueTypes.QUANTITY);

                const hasPeopleCount = values.some(v => v.type === this.valueTypes.PEOPLE_COUNT);

                const hasDiscount = values.some(v => v.type === this.valueTypes.DISCOUNT_PERCENT);

                console.log('判定結果:', { hasWallet, hasUnitPrice, hasQuantity, hasPeopleCount, hasDiscount });

                // おつり計算の判定

                if (/おつり|お釣り|釣り/.test(input)) {

                    console.log('計算タイプ: おつり計算');

                    return this.calculationTypes.CHANGE_CALCULATION;

                }

                

                // 割り勘計算の判定

                if (/割り勘|分ける|シェア/.test(input) && hasPeopleCount) {

                    console.log('計算タイプ: 割り勘計算');

                    return this.calculationTypes.SPLIT_BILL;

                }

                

                // 割引計算の判定

                if (hasDiscount) {

                    console.log('計算タイプ: 割引計算');

                    return this.calculationTypes.DISCOUNT_PRICE;

                }

                

                // 購入計算の判定（単価×個数）

                if (hasUnitPrice && hasQuantity) {

                    if (hasWallet) {

                        console.log('計算タイプ: 残金計算（財布あり）');

                        return this.calculationTypes.CHANGE_CALCULATION; // おつり計算として処理

                    } else {

                        console.log('計算タイプ: 購入合計');

                        return this.calculationTypes.PURCHASE_TOTAL;

                    }

                }

                

                // 合計計算の判定

                if (/合計|総額|全部/.test(input)) {

                    console.log('計算タイプ: 合計計算');

                    return this.calculationTypes.TOTAL_SUM;

                }

                

                // 財布がある場合の残金計算

                if (hasWallet) {

                    console.log('計算タイプ: 残金計算');

                    return this.calculationTypes.REMAINING_MONEY;

                }

                

                console.log('計算タイプ: デフォルト（購入合計）');

                return this.calculationTypes.PURCHASE_TOTAL;

            }

            // 計算実行（モード別性能差実装）

            calculate(input, mode = 'normal') {

                const startTime = Date.now();

                

                try {

                    let detectedValues, calculationType, result, confidence;

                    switch (mode) {

                        case 'speed':

                            detectedValues = this.detectValuesSpeed(input);

                            calculationType = this.determineCalculationTypeSpeed(input, detectedValues);

                            result = this.executeCalculationSpeed(calculationType, detectedValues);

                            confidence = 0.75;

                            break;

                        case 'thinking':

                            detectedValues = this.detectValuesThinking(input);

                            const allPossibleTypes = this.getAllPossibleCalculationTypes(input, detectedValues);

                            const allResults = this.executeAllCalculationPatterns(allPossibleTypes, detectedValues, input);

                            const bestResult = this.selectBestResult(allResults, input);

                            

                            calculationType = bestResult.type;

                            result = bestResult.result;

                            confidence = this.calculateConfidenceThinking(calculationType, detectedValues, input, allResults);

                            break;

                        default: // normal

                            detectedValues = this.detectValues(input);

                            calculationType = this.determineCalculationType(input, detectedValues);

                            result = this.executeCalculation(calculationType, detectedValues, input);

                            confidence = this.calculateConfidence(calculationType, detectedValues, input);

                            break;

                    }

                    

                    if (detectedValues.length === 0) {

                        throw new Error('計算に必要な数値が見つかりません');

                    }

                    

                    const processingTime = Date.now() - startTime;

                    

                    return {

                        ...result,

                        detectedValues: detectedValues,

                        calculationType: calculationType,

                        confidence: confidence,

                        processingTime: processingTime,

                        mode: mode,

                        thinkingDetails: mode === 'thinking' ? this.getThinkingDetails() : null

                    };

                } catch (error) {

                    return {

                        formula: 'CalcAI解析エラー',

                        answer: 'エラー',

                        answerType: 'エラーが発生しました',

                        detectedValues: [],

                        calculationType: 'error',

                        confidence: 0.0,

                        processingTime: Date.now() - startTime,

                        mode: mode,

                        error: error.message

                    };

                }

            }

            // 計算実行

            executeCalculation(type, values, input) {

                switch (type) {

                    case this.calculationTypes.PURCHASE_TOTAL:

                        return this.calculatePurchaseTotal(values);

                    case this.calculationTypes.CHANGE_CALCULATION:

                        return this.calculateChange(values);

                    case this.calculationTypes.REMAINING_MONEY:

                        return this.calculateRemainingMoney(values);

                    case this.calculationTypes.SPLIT_BILL:

                        return this.calculateSplitBill(values);

                    case this.calculationTypes.DISCOUNT_PRICE:

                        return this.calculateDiscountPrice(values);

                    case this.calculationTypes.TOTAL_SUM:

                        return this.calculateTotalSum(values);

                    default:

                        return this.calculatePurchaseTotal(values);

                }

            }

            // 購入合計計算

            calculatePurchaseTotal(values) {

                const unitPrice = values.find(v => v.type === this.valueTypes.UNIT_PRICE) ||

                                values.find(v => v.currency && v.value < 10000);

                const quantity = values.find(v => v.type === this.valueTypes.QUANTITY) ||

                               values.find(v => !v.currency && v.value < 100);

                if (!unitPrice || !quantity) {

                    throw new Error('単価または個数が特定できません');

                }

                const total = unitPrice.value * quantity.value;

                const currency = unitPrice.currency || '¥';

                return {

                    formula: `${unitPrice.original} × ${quantity.value} = ${currency}${total.toLocaleString()}`,

                    answer: `${currency}${total.toLocaleString()}`,

                    answerType: '購入合計金額'

                };

            }

            // おつり計算（個数単価問題対応版）

            calculateChange(values) {

                console.log('=== おつり計算開始 ===');

                console.log('値:', values);

                

                const wallet = values.find(v => v.type === this.valueTypes.WALLET_MONEY);

                const unitPrice = values.find(v => v.type === this.valueTypes.UNIT_PRICE);

                const quantity = values.find(v => v.type === this.valueTypes.QUANTITY);

                if (!wallet) {

                    throw new Error('財布の金額が特定できません');

                }

                let totalCost;

                let formula;

                

                // 単価×個数の場合

                if (unitPrice && quantity) {

                    totalCost = unitPrice.value * quantity.value;

                    const change = wallet.value - totalCost;

                    const currency = wallet.currency || '¥';

                    

                    formula = `${wallet.original} - (${unitPrice.original} × ${quantity.value}) = ${currency}${change.toLocaleString()}`;

                    

                    console.log('単価×個数計算:', {

                        wallet: wallet.value,

                        unitPrice: unitPrice.value,

                        quantity: quantity.value,

                        totalCost: totalCost,

                        change: change

                    });

                    

                    return {

                        formula: formula,

                        answer: `${currency}${change.toLocaleString()}`,

                        answerType: 'おつり'

                    };

                }

                

                // 通常のおつり計算

                const cost = values.find(v => v !== wallet && v.currency);

                if (!cost) {

                    throw new Error('費用が特定できません');

                }

                const change = wallet.value - cost.value;

                const currency = wallet.currency || cost.currency || '¥';

                return {

                    formula: `${wallet.original} - ${cost.original} = ${currency}${change.toLocaleString()}`,

                    answer: `${currency}${change.toLocaleString()}`,

                    answerType: 'おつり'

                };

            }

            // 残金計算

            calculateRemainingMoney(values) {

                const unitPrice = values.find(v => v.type === this.valueTypes.UNIT_PRICE) ||

                                values.find(v => v.currency && v.value < 10000);

                const quantity = values.find(v => v.type === this.valueTypes.QUANTITY) ||

                               values.find(v => !v.currency && v.value < 100);

                const wallet = values.find(v => v.type === this.valueTypes.WALLET_MONEY) ||

                             values.reduce((max, v) => v.value > max.value ? v : max);

                if (!unitPrice || !quantity || !wallet) {

                    throw new Error('必要な値が特定できません');

                }

                const totalCost = unitPrice.value * quantity.value;

                const remaining = wallet.value - totalCost;

                const currency = wallet.currency || unitPrice.currency || '¥';

                return {

                    formula: `${wallet.original} - (${unitPrice.original} × ${quantity.value}) = ${currency}${remaining.toLocaleString()}`,

                    answer: `${currency}${remaining.toLocaleString()}`,

                    answerType: '残金'

                };

            }

            // 割り勘計算

            calculateSplitBill(values) {

                const total = values.find(v => v.currency) || values[0];

                const people = values.find(v => v.type === this.valueTypes.PEOPLE_COUNT) ||

                             values.find(v => !v.currency && v.value < 20);

                if (!total || !people) {

                    throw new Error('総額または人数が特定できません');

                }

                const perPerson = total.value / people.value;

                const currency = total.currency || '¥';

                return {

                    formula: `${total.original} ÷ ${people.value}人 = ${currency}${perPerson.toLocaleString()}`,

                    answer: `${currency}${perPerson.toLocaleString()}`,

                    answerType: '1人あたりの金額'

                };

            }

            // 割引計算

            calculateDiscountPrice(values) {

                const originalPrice = values.find(v => v.currency) || values[0];

                const discountPercent = values.find(v => v.type === this.valueTypes.DISCOUNT_PERCENT);

                if (!originalPrice || !discountPercent) {

                    throw new Error('元の価格または割引率が特定できません');

                }

                const discountAmount = originalPrice.value * (discountPercent.value / 100);

                const finalPrice = originalPrice.value - discountAmount;

                const currency = originalPrice.currency || '¥';

                return {

                    formula: `${originalPrice.original} - (${originalPrice.original} × ${discountPercent.value}%) = ${currency}${finalPrice.toLocaleString()}`,

                    answer: `${currency}${finalPrice.toLocaleString()}`,

                    answerType: '割引後価格'

                };

            }

            // 合計計算

            calculateTotalSum(values) {

                const moneyValues = values.filter(v => v.currency || v.value > 100);

                const total = moneyValues.reduce((sum, v) => sum + v.value, 0);

                const currency = moneyValues.find(v => v.currency)?.currency || '¥';

                return {

                    formula: moneyValues.map(v => v.original).join(' + ') + ` = ${currency}${total.toLocaleString()}`,

                    answer: `${currency}${total.toLocaleString()}`,

                    answerType: '合計金額'

                };

            }

            // スピードモード専用メソッド群（Normalベース簡略化）

            detectValuesSpeed(input) {

                console.log('=== Speed Mode: 簡略化検出開始 ===');

                

                // Normalモードの検出ロジックを使用（最初の3つまで）

                const fullValues = this.detectValues(input);

                const speedValues = fullValues.slice(0, 3);

                

                console.log('Speed検出結果:', speedValues);

                return speedValues;

            }

            determineCalculationTypeSpeed(input, values) {

                // Normalモードと同じロジックを使用

                return this.determineCalculationType(input, values);

            }

            executeCalculationSpeed(type, values) {

                // Normalモードと同じ計算ロジックを使用

                return this.executeCalculation(type, values, '');

            }

            // Thinkingモード専用メソッド群（Normalベース拡張）

            detectValuesThinking(input) {

                console.log('=== Thinking Mode: 拡張検出開始 ===');

                

                // Normalモードの検出結果をベースに使用

                const baseValues = this.detectValues(input);

                

                // 安全な拡張処理のみ追加

                baseValues.forEach((val, index) => {

                    try {

                        // 文脈スコア計算（安全版）

                        val.contextScore = this.calculateSafeContextScore(val.context, input);

                        

                        // 位置スコア計算

                        val.positionScore = val.position / input.length;

                        

                        // 代替タイプ候補（安全版）

                        val.alternativeTypes = this.getSafeAlternativeTypes(val.type, val.value);

                        

                        // 信頼度計算

                        val.confidenceScore = (val.contextScore + (1 - val.positionScore)) / 2;

                        

                        console.log(`Thinking拡張 ${index + 1}:`, {

                            value: val.value,

                            type: val.type,

                            contextScore: val.contextScore,

                            confidenceScore: val.confidenceScore

                        });

                        

                    } catch (error) {

                        console.warn('Thinking拡張処理エラー:', error);

                        // エラーが発生してもベース値は保持

                        val.contextScore = 0.5;

                        val.positionScore = 0.5;

                        val.alternativeTypes = [];

                        val.confidenceScore = 0.5;

                    }

                });

                

                console.log('Thinking最終結果:', baseValues);

                return baseValues;

            }

            // 安全な文脈スコア計算

            calculateSafeContextScore(context, fullInput) {

                try {

                    const keywords = ['円', '¥', '個', 'つ', '財布', '買', '購入', 'おつり'];

                    const contextLower = (context || '').toLowerCase();

                    const inputLower = (fullInput || '').toLowerCase();

                    

                    let score = 0.3; // ベーススコア

                    

                    keywords.forEach(keyword => {

                        if (contextLower.includes(keyword) || inputLower.includes(keyword)) {

                            score += 0.1;

                        }

                    });

                    

                    return Math.min(score, 1.0);

                } catch (error) {

                    return 0.5; // デフォルト値

                }

            }

            // 安全な代替タイプ取得

            getSafeAlternativeTypes(currentType, value) {

                try {

                    const alternatives = [];

                    

                    if (value < 100) {

                        alternatives.push('quantity', 'people_count');

                    }

                    if (value > 100) {

                        alternatives.push('total_money', 'wallet_money', 'unit_price');

                    }

                    if (currentType === 'unit_price') {

                        alternatives.push('total_money');

                    }

                    

                    return alternatives.filter(alt => alt !== currentType);

                } catch (error) {

                    return [];

                }

            }

            // Thinking専用：安全な計算タイプ評価

            getAllPossibleCalculationTypes(input, values) {

                console.log('=== Thinking: 計算タイプ評価開始 ===');

                

                try {

                    const possibleTypes = [];

                    

                    // 基本的な計算タイプを安全に評価

                    const basicTypes = [

                        this.calculationTypes.CHANGE_CALCULATION,

                        this.calculationTypes.PURCHASE_TOTAL,

                        this.calculationTypes.REMAINING_MONEY,

                        this.calculationTypes.TOTAL_SUM

                    ];

                    

                    basicTypes.forEach(type => {

                        try {

                            const score = this.evaluateCalculationTypeScoreSafe(type, input, values);

                            if (score > 0.1) {

                                possibleTypes.push({ type, score });

                                console.log(`タイプ評価: ${type} = ${score}`);

                            }

                        } catch (error) {

                            console.warn(`タイプ評価エラー ${type}:`, error);

                        }

                    });

                    

                    const sorted = possibleTypes.sort((a, b) => b.score - a.score);

                    console.log('評価結果:', sorted);

                    return sorted;

                    

                } catch (error) {

                    console.error('計算タイプ評価エラー:', error);

                    // フォールバック：Normalモードの判定を使用

                    const fallbackType = this.determineCalculationType(input, values);

                    return [{ type: fallbackType, score: 0.7 }];

                }

            }

            // 安全な計算タイプスコア評価

            evaluateCalculationTypeScoreSafe(type, input, values) {

                try {

                    const typeKeywords = {

                        [this.calculationTypes.PURCHASE_TOTAL]: ['買', '購入', '商品', '×', 'かける'],

                        [this.calculationTypes.CHANGE_CALCULATION]: ['おつり', 'お釣り', '釣り', '支払', 'から'],

                        [this.calculationTypes.REMAINING_MONEY]: ['残', '財布', '所持', 'あと', '余り'],

                        [this.calculationTypes.SPLIT_BILL]: ['割り勘', '分ける', 'シェア', '人で', '÷'],

                        [this.calculationTypes.DISCOUNT_PRICE]: ['割引', 'セール', 'オフ', '%', 'パーセント'],

                        [this.calculationTypes.TOTAL_SUM]: ['合計', '総額', '全部', '+', '足す']

                    };

                    const keywords = typeKeywords[type] || [];

                    const inputLower = (input || '').toLowerCase();

                    

                    let matches = 0;

                    keywords.forEach(keyword => {

                        if (inputLower.includes(keyword)) {

                            matches++;

                        }

                    });

                    

                    const baseScore = matches / Math.max(keywords.length, 1);

                    

                    // 値の存在による補正

                    let valueBonus = 0;

                    if (type === this.calculationTypes.CHANGE_CALCULATION) {

                        const hasWallet = values.some(v => v.type === 'wallet_money');

                        const hasPrice = values.some(v => v.type === 'unit_price' || v.currency);

                        if (hasWallet && hasPrice) valueBonus = 0.3;

                    }

                    

                    return Math.min(baseScore + valueBonus, 1.0);

                    

                } catch (error) {

                    return 0.2; // デフォルトスコア

                }

            }

            executeAllCalculationPatterns(possibleTypes, values, input) {

                console.log('=== Thinking: 全パターン実行開始 ===');

                

                return possibleTypes.map(({ type, score }) => {

                    try {

                        console.log(`パターン実行: ${type}`);

                        const result = this.executeCalculation(type, values, input);

                        console.log(`パターン成功: ${type}`, result);

                        

                        return {

                            type,

                            score,

                            result,

                            success: true

                        };

                    } catch (error) {

                        console.warn(`パターンエラー ${type}:`, error.message);

                        return {

                            type,

                            score: score * 0.1,

                            result: null,

                            success: false,

                            error: error.message

                        };

                    }

                });

            }

            selectBestResult(allResults, input) {

                console.log('=== Thinking: 最適解選択 ===');

                console.log('全結果:', allResults);

                

                const successfulResults = allResults.filter(r => r.success);

                if (successfulResults.length === 0) {

                    console.error('全パターンでエラー発生');

                    throw new Error('全ての計算パターンでエラーが発生しました');

                }

                

                const bestResult = successfulResults.reduce((best, current) =>

                    current.score > best.score ? current : best

                );

                

                console.log('選択された最適解:', bestResult);

                return bestResult;

            }

            calculateConfidenceThinking(calculationType, values, input, allResults) {

                try {

                    // ベース信頼度

                    let confidence = this.calculateConfidence(calculationType, values, input);

                    

                    // 成功率による補正

                    const successRate = allResults.filter(r => r.success).length / Math.max(allResults.length, 1);

                    confidence *= (0.5 + successRate * 0.5);

                    

                    // 文脈スコアによる補正（安全版）

                    const contextScores = values.map(v => v.contextScore || 0.5);

                    const avgContextScore = contextScores.reduce((sum, score) => sum + score, 0) / Math.max(contextScores.length, 1);

                    confidence = (confidence + avgContextScore) / 2;

                    

                    // Thinkingモードボーナス

                    confidence = Math.min(confidence + 0.05, 0.99);

                    

                    return confidence;

                    

                } catch (error) {

                    console.warn('Thinking信頼度計算エラー:', error);

                    return 0.75; // デフォルト値

                }

            }

            getThinkingDetails() {

                return {

                    analyzedPatterns: 'multiple',

                    verificationSteps: 6,

                    confidenceFactors: ['context', 'position', 'alternatives', 'verification', 'quantity_conflict_resolution']

                };

            }

            // 補助メソッド群

            analyzeContextScore(context, fullInput) {

                const keywords = ['円', '¥', '個', 'つ', '財布', '買', '購入'];

                const matches = keywords.filter(k => context.includes(k)).length;

                return Math.min(matches / keywords.length + 0.3, 1.0);

            }

            analyzePositionScore(position, totalLength) {

                return 1.0 - (position / totalLength) * 0.3;

            }

            getAlternativeTypes(context, value) {

                const alternatives = [];

                if (value < 100) alternatives.push('quantity', 'people_count');

                if (value > 100) alternatives.push('total_money', 'wallet_money');

                return alternatives;

            }

            evaluateCalculationTypeScore(type, input, values) {

                const typeKeywords = {

                    [this.calculationTypes.PURCHASE_TOTAL]: ['買', '購入', '商品', '×', 'かける'],

                    [this.calculationTypes.CHANGE_CALCULATION]: ['おつり', 'お釣り', '釣り', '支払', 'から'],

                    [this.calculationTypes.REMAINING_MONEY]: ['残', '財布', '所持', 'あと', '余り'],

                    [this.calculationTypes.SPLIT_BILL]: ['割り勘', '分ける', 'シェア', '人で', '÷'],

                    [this.calculationTypes.DISCOUNT_PRICE]: ['割引', 'セール', 'オフ', '%', 'パーセント'],

                    [this.calculationTypes.TOTAL_SUM]: ['合計', '総額', '全部', '+', '足す']

                };

                const keywords = typeKeywords[type] || [];

                const matches = keywords.filter(k => input.includes(k)).length;

                return matches / Math.max(keywords.length, 1);

            }

            // 信頼度計算

            calculateConfidence(calculationType, values, input) {

                let confidence = 0.5;

                // 個数問題解決による信頼度向上

                const hasResolvedQuantityConflict = values.some(v => v.resolvedAs);

                if (hasResolvedQuantityConflict) confidence += 0.15;

                // 値の明確性

                const hasTypedValues = values.some(v => v.type !== 'unknown');

                if (hasTypedValues) confidence += 0.2;

                // 通貨記号の存在

                const hasCurrency = values.some(v => v.currency);

                if (hasCurrency) confidence += 0.1;

                // 計算タイプの明確性

                const typeKeywords = {

                    [this.calculationTypes.PURCHASE_TOTAL]: ['買', '購入', '商品'],

                    [this.calculationTypes.CHANGE_CALCULATION]: ['おつり', 'お釣り', '釣り'],

                    [this.calculationTypes.REMAINING_MONEY]: ['残', '財布', '所持'],

                    [this.calculationTypes.SPLIT_BILL]: ['割り勘', '分ける', 'シェア'],

                    [this.calculationTypes.DISCOUNT_PRICE]: ['割引', 'セール', 'オフ'],

                    [this.calculationTypes.TOTAL_SUM]: ['合計', '総額', '全部']

                };

                const keywords = typeKeywords[calculationType] || [];

                const hasKeywords = keywords.some(keyword => input.includes(keyword));

                if (hasKeywords) confidence += 0.15;

                // 値の数と妥当性

                if (values.length >= 2) confidence += 0.05;

                return Math.min(confidence, 0.98);

            }

        }
